    28	  return {

    33	  } as T & ReportStatusFields

    58	  const reportStatusFields = new Set([
    59	    'reportStatusCode',
    60	    'reportStatusLabel',
    61	    'reportedAt',
    62	  ])
    63	  const onlyUpdatesReportStatus = Object.keys(updates).every((field) =>
    64	    reportStatusFields.has(field),
    65	  )
    66	  const reportStatusUpdates = updates as Partial<ReportStatusFields>
    67	  const removesReportedMark =
    68	    reportStatusUpdates.reportStatusCode === 'pending' &&
    69	    reportStatusUpdates.reportStatusLabel === 'Pendiente' &&
    70	    reportStatusUpdates.reportedAt === undefined
    71	
    72	  if (!onlyUpdatesReportStatus || !removesReportedMark) {
    73	    throw new Error(REPORTED_RECORD_IMMUTABLE_MESSAGE)
    74	  }
    75	}
    76	
    77	export function markAsReported<T extends Partial<ReportStatusFields>>(record: T) {
    78	  return {
    79	    ...normalizeReportStatus(record),
    80	    reportStatusCode: 'reported' as const,
    81	    reportStatusLabel: 'Reportado' as const,
    82	    reportedAt: new Date().toISOString(),
    83	  } as T & ReportStatusFields
    84	}
    85	
    86	export function markAsPending<T extends Partial<ReportStatusFields>>(record: T) {
    87	  return {
    88	    ...normalizeReportStatus(record),
    89	    reportStatusCode: 'pending' as const,
    90	    reportStatusLabel: 'Pendiente' as const,
    91	    reportedAt: undefined,
    92	  } as T & ReportStatusFields
    93	}
